from math import e

def sigmoid(x):
    return 1 / (1 + np.pow(e, -x))

x = np.linspace(-5, 5, 100)
y = sigmoid(x)

plt.plot(x, y)
plt.ylim(0, 1)
plt.show()

import numpy as np
import matplotlib.pyplot as plt

v = np.array([0, 1])
a = 1
M = np.matrix([[np.cos(a), -np.sin(a)], 
               [np.sin(a),  np.cos(a)]])
vs = [v]
for i in range(100):
    v = np.array(v * M * 0.99).flatten()
    vs.append(v)
    
vs = np.array(vs)

plt.figure(figsize=(6, 6))
plt.xlim(-1, 1)
plt.ylim(-1, 1)

plt.plot(*vs.T)
plt.show()

Одним з алгоритмів навчання з учителем є так звана поліноміальна регресія. Для цього методу, потрібно маючи матрицю $$X=\begin{bmatrix} x_{1}\\ x_{2}\\ \vdots \\ x_{m} \end{bmatrix}$$ отримати матрицю ступенів: $$X_p=\begin{bmatrix} x^{0}_{1} & x^{1}_{1} & \dotsc & x^{p}_{1}\\\ x^{0}_{2} & x^{1}_{2} & \dotsc & x^{p}_{2}\\\ \vdots & \vdots & & \vdots \\ x^{0}_{m} & x^{1}_{m} & \dotsc & x^{p}_{m} \end{bmatrix}$$

Маючи матрицю $X$, задану нижче, обчисліть матрицю $X_p$, для $p=5$. Розв'язати задачу як з використанням циклу за степенями, так і без нього.

Корисні функції: stack, concatenate.

X = np.linspace(-2.2, 2.2, 1000)[:, np.newaxis]
print(f'Shape of X: {X.shape}')

p = 5


X_p = np.matrix([[np.pow(x, i) for i in range(p + 1)] for x in X.flatten()])
print(f'Shape of X_p {X_p.shape}')

assert X_p.shape == (1000, 6)

plt.plot(
    X[..., 0],
    X_p @ np.array([[0, 4, 0, -5, 0, 1]]).T
)
plt.show()

from math import pi

def rastrigin(x, y):
    return 20 + (np.pow(x, 2) - 10 * np.cos(2 * pi * x)) + (np.pow(y, 2) - 10 * np.cos(2 * pi * y))

x = np.linspace(-5, 5, 500)
y = np.linspace(-5, 5, 500)

[X, Y] = np.meshgrid(x,y)

Z = rastrigin(X, Y)

plt.figure(figsize=(10, 10))
plt.contourf(X, Y, Z, levels=10)
plt.show()
